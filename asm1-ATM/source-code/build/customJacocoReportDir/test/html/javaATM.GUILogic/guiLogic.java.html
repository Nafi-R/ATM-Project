package javaATM.GUILogic;

import java.time.LocalDate;
import java.util.List;

import javaATM.Card;

public class guiLogic {
    public static Card validCard;
    
    public static boolean checkValid(int cardNo, int pinNo, List<Card> cardsList) {
        //check if card is valid
        //if not valid, decrement attempts
        boolean cardisValid = false;

        for (int i=0; i<cardsList.size(); i++) {
            Card card = cardsList.get(i);
            System.out.println(card.getCardNumber() + " " + card.getPin());
            if (cardNo == card.getCardNumber() && pinNo == card.getPin()) {
                if (card.getStatus().equals("Valid")) {
                    cardisValid = true;
                    validCard = card;
                    break;
                }
            }
        }
        return cardisValid;
    }

    public static boolean isDateValid(LocalDate issue, LocalDate expiry) {
        int compareValIssue = issue.compareTo(LocalDate.now());
        int compareValExpiry = expiry.compareTo(LocalDate.now());
        System.out.println(issue+ " "  + LocalDate.now());
        if(compareValIssue < 0 && compareValExpiry > 0) {
            return true;
        }
        return false;
    }
}
